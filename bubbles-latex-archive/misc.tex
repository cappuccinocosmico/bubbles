\section{Exercises}

\begin{Exercise}
    Given the trigonometric definitions of $\sin$ and $\cos$ derive
    \begin{align*}
        \sin(a+b)&=\sin(a)\cos(b) + \cos(a) \sin(b)\\
        \cos(a+b)&=\cos(a)\cos(b) - \sin(a) \sin(b)\\
    \end{align*}
\end{Exercise}
\begin{Answer}
    \begin{align*}
        \sin(a+b) &= \frac{e^{i(a+b)}-e^{-ix(a+b)}}{2i}\\
        &= \frac{e^{ia}e^{ib}-e^{-ia}e^{-ib}{2i}\\
    \end{align*}
\end{Answer}
